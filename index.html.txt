<script>
  let pdfViewer = document.getElementById("pdfViewer");
  let appText = document.getElementById("appText");
  let errorMsg = document.getElementById("errorMsg");
  let appInput = document.getElementById("appInput");

  let validApp = "UK26ES0300045530";

  function searchApp() {
    let appNo = appInput.value;
    loadCertificate(appNo);
  }

  function loadCertificate(appNo) {
    if (appNo === validApp) {
      appText.innerText = "Application No: " + appNo;
      pdfViewer.src = "certificate.pdf";
      errorMsg.innerText = "";
    } else {
      appText.innerText = "";
      pdfViewer.src = "";
      errorMsg.innerText = "Certificate not found (404)";
    }
  }

  // ✅ AUTO LOAD + AUTO FILL FIXED
  window.onload = function () {
    let autoApp = window.location.hash.substring(1);

    if (autoApp) {
      appInput.value = autoApp;     // shows in input
      loadCertificate(autoApp);     // loads PDF
    }
  };
</script>
